## Code

```bash
tmppath="/tmp/$(date +%F)/ansible-roles"
mkdir -p $tmppath
for d in $(find $PWD -mindepth 1 -maxdepth 1 -type d); do
  echo "$d"
  savename="${d//${PWD}\//}"
  find "$d" -type f -name "*.yml" -print -exec cat {} \; | sed -e 's@'"$(echo "$d")"'@'"$(echo "\n# file: $d")"'@g' | tee "$tmppath/${savename//[^a-zA-Z0-9.]/_}"
done
```
