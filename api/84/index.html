    "printcommands() {",
    "  for i in $(echo \"$PATH\" | sed 's/:/\\n/g' | grep -v ' '); do",
    "    echo \"#######################################################\"",
    "    echo \"$i\"",
    "    echo \"#######################################################\"",
    "    echo \"$(find $i -maxdepth 2 -type f ! -name '*.dll' 2>&1 | grep -E \"\\/[a-z0-9]*$|*.exe$\" | grep -v license)\"",
    "    echo \"\"",
    "  done",
    "}",
    "",
    "printcommands | tee /tmp/window-commands.txt"
